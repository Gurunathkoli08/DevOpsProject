pipeline {
    agent any

    stages {
        stage('Clone Repository') {
            steps {
                git 'https://github.com/Gurunathkoli08/DevOpsProject.git'
            }
        }

        stage('Build WAR (Optional)') {
            steps {
                echo "WAR already exported manually from Eclipse"
            }
        }

        stage('Build Docker Image') {
            steps {
                sh 'docker build -t java-static-site .'
            }
        }

        stage('Run Docker Container') {
            steps {
                sh '''
                    docker stop my-static-site || true
                    docker rm my-static-site || true
                    docker run -d -p 8085:8090 --name my-static-site java-static-site
                '''
            }
        }

        stage('(Optional) Deploy with Ansible') {
            steps {
                sh 'ansible-playbook ansible/deploy.yml || echo "Ansible step skipped"'
            }
        }
    }
}
